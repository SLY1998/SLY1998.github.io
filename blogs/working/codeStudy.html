    # 3、uni.getUpdateManager() 管理小程序更新

    • uni.canIUse(String) 判断应用的 API，回调，参数，组件是否在当前版本可用。
    • uni.getUpdateManager() 本 API 返回全局唯一的版本更新管理器对象： updateManager，用于管理小程序更新。
    • updateManager 对象的方法列表：
    1. onCheckForUpdate ; callback ; 当向小程序后台请求完新版本信息，会进行回调
    2. onUpdateReady ; callback ; 当新版本下载完成，会进行回调
    3. onUpdateFailed ; callback ; 当新版本下载失败，会进行回调
    4. applyUpdate 当新版本下载完成，调用该方法会强制当前小程序应用上新版本并重启

    在 onShow()中调用

    // 获取小程序更新机制兼容
if (uni.canIUse("getUpdateManager")) {
  let _this = this;
  const updateManager = uni.getUpdateManager();
  // 检查是否有新版本发布
  updateManager.onCheckForUpdate(function (res) {
    console.log(res.hasUpdate); // true
    if (res.hasUpdate) {
      //小程序有新版本，则静默下载新版本，做好更新准备
      updateManager.onUpdateReady(function () {
        uni.showModal({
          title: "更新提示",
          content: "新版本已经准备好，是否重启应用？",
          success: function (res) {
            if (res.confirm) {
              //新的版本已经下载好，调用 applyUpdate 应用新版本并重启
              updateManager.applyUpdate();
            } else if (res.cancel) {
              //如果需要强制更新，则给出二次弹窗，如果不需要，则这里的代码都可以删掉了
              uni.showModal({
                title: "温馨提示",
                content:
                  "本次版本更新涉及到新的功能添加，旧版本无法正常访问的哦~",
                showCancel: false, //隐藏取消按钮，也可显示，取消会走res.cancel，然后从新开始提示
                success: function (res) {
                  //第二次提示后，强制更新
                  if (res.confirm) {
                    // 新的版本已经下载好，调用 applyUpdate 应用新版本并重启
                    updateManager.applyUpdate();
                  } else if (res.cancel) {
                    //重新回到版本更新提示
                    _this.autoUpdate();
                  }
                },
              });
            }
          },
        });
      });
      // 新的版本下载失败
      updateManager.onUpdateFailed(function () {
        uni.showModal({
          title: "温馨提示",
          content: "新版本已经上线，请您删除当前小程序，重新搜索打开",
        });
      });
    }
  });
} else {
  // 提示用户在最新版本的客户端上体验
  uni.showModal({
    title: "温馨提示",
    content: "当前微信版本过低，可能无法使用该功能，请升级到最新版本后重试。",
  });
}

    # 4、css 属性：env()和 constant()设置安全区域

    env()和 constant()，是 IOS11 新增特性，Webkit 的 css 函数，用于设定安全区域与边界的距离，有 4 个预定义变量：

    • safe-area-inset-left：安全区域距离左边边界的距离
    • safe-area-inset-right：安全区域距离右边边界的距离
    • safe-area-inset-top：安全区域距离顶部边界的距离
    • safe-area-inset-bottom ：安全距离底部边界的距离

    env()和 constant()函数有个必要的使用前提，H5 网页设置 viewport-fit=cover 的时候才生效，小程序里的 viewport-fit 默认是 cover。

    page {
  padding-bottom: calc(constant(safe-area-inset-bottom) + 150rpx);
  padding-bottom: calc(env(safe-area-inset-bottom) + 150rpx);
}

/* not 表示不支持括号内的属性 */
@supports not (constant(safe-area-inset-bottom)) {
  page {
    padding-bottom: 150rpx;
  }
}

    # 7、小程序可拖动按钮组件

    <template>
  <cover-view
    ref="dragDom"
    class="drag-content"
    @click.stop="dragClick"
    @touchstart="touchstart"
    @touchendt="touchend"
    @touchmove.capture.stop.prevent="touchmove"
    :style="{
      left: `${left}px`,
      top: `${top}px`,
      backgroundColor: 'transparent',
      width: `${dragSize}rpx`,
      height: `${dragSize}rpx`,
    }">
    <cover-view class="drag-stop">
      <cover-image
        :src="image"
        mode="widthFix"
        v-if="image"></cover-image>
    </cover-view>
  </cover-view>
</template>

<script setup lang="ts">
import { reactive, toRefs, defineEmits, computed, onMounted } from "vue";
import store from "../store";

const emit = defineEmits(["dragClick"]);

const data = reactive({
  left: -100,
  top: -100,
  isdrag: false,
  NowLeft: 0,
  disX: 0,
  NowTop: 0,
  disY: 0,
  domWidth: 0,
  minTop:
    store.getters.navData.menuTop - 20 + store.getters.navData.navBarHeight,
});

const props = defineProps({
  dragSize: {
    type: String,
    default: "107",
  },
  image: {
    type: String,
    default: "",
  },
});

// 初始化按钮位置
onMounted(() => {
  const systemInfo = uni.getSystemInfoSync();
  // px = rpx / 750 * wx.getSystemInfoSync().windowWidth；rpx = px * 750 / wx.getSystemInfoSync().windowWidth
  data.domWidth = (props.dragSize / 750) * systemInfo.windowWidth;
  // 默认距离顶部的距离 = 手机的可用高度 - 底部tab拦高度 - 自身高度
  data.top =
    systemInfo.windowHeight -
    ((150 / 750) * systemInfo.windowWidth + systemInfo.safeAreaInsets.bottom) -
    data.domWidth;
  // 默认距离左侧的距离 = 手机的可用宽度 - 自身高度
  data.left = systemInfo.windowWidth - data.domWidth;
});

// 按钮点击事件
const dragClick = () => {
  emit("dragClick");
};

const touchstart = (e) => {
  data.isdrag = true;
  data.NowLeft = data.left;
  data.NowTop = data.top;
  data.disX = e.touches[0].pageX;
  data.disY = e.touches[0].pageY;
  return false;
};

const touchend = () => {
  data.isdrag = false;
};

const touchmove = (e) => {
  e.preventDefault();
  if (data.isdrag) {
    // 获取系统信息
    const systemInfo = uni.getSystemInfoSync();
    data.left = data.NowLeft + e.touches[0].pageX - data.disX;
    // 触及左侧
    if (data.left < 0) {
      data.left = 0;
    }
    // 触及右侧
    const maxLeft = systemInfo.windowWidth - data.domWidth;
    if (data.left > maxLeft) {
      data.left = maxLeft;
    }
    data.top = data.NowTop + e.touches[0].pageY - data.disY;
    // 触及顶部
    if (data.top < data.minTop) {
      data.top = data.minTop;
    }
    // 触及底部
    const maxTop =
      systemInfo.windowHeight -
      ((150 / 750) * systemInfo.windowWidth +
        systemInfo.safeAreaInsets.bottom) -
      data.domWidth;
    if (data.top > maxTop) {
      data.top = maxTop;
    }
    return false;
  }
};

const { dragSize, image } = toRefs(props);
const { left, top } = toRefs(data);
</script>

<style lang="less" scoped>
.drag-content {
  width: 120rpx;
  height: 120rpx;
  border-radius: 50%;
  position: fixed;
  z-index: 999;
}

.drag-content:active {
  transform: scale(0.95);
  transition: all 0.2s;
}
.drag-stop {
  width: 100%;
  height: 100%;
}

.drag-content image {
  width: 100%;
}
</style>

    # pdf 文件流解析下载

    后台接口返回文件流，前端收到数据并解析成文件下载

    // 1.接口请求配置项加入：getResponse: true
// 2.正常调接口
API(params).then(async (res) => {
  // 错误处理
  if (res.data?.code) {
    message({
      msg: res.data?.text,
      type: "error",
    });
    return;
  } else {
    // 解析成blob
    const temp = await res.response.blob();
    // 生成想要的文件格式，type看需求而定
    // type，默认值为 ""，它代表了将会被放入到 blob 中的数组内容的 MIME 类型。
    const blob = new Blob([temp], { type: "application/pdf" });
    // 生成想要的文件格式url
    const objectURL = URL.createObjectURL(blob);
    // 创建a链接下载文件
    const btn = document.createElement("a");
    btn.download = "文件名.pdf";
    btn.href = objectURL;
    btn.click();
    // 释放URL 对象
    URL.revokeObjectURL(objectURL);
  }
});

    # 前端上传、解析 excel 文件

    结合 antd Upload 组件实现

    import XLSX from "xlsx";
// 配置项
const props = {
  multiple: false,
  maxCount: 1,
  showUploadList: false,
  beforeUpload(file) {
    const fileType = file.name.split(".").pop();
    if (fileType !== "xlsx" && fileType !== "xls") {
      message({
        type: "warning",
        msg: "只能选择excel文件导入",
      });
      return false;
    }
    setUploadLoading(true);
    return true;
  },
  onChange(info) {
    const { status, name } = info.file;
    if (status === "done") {
      let fileReader = new FileReader();
      // 解析数据
      fileReader.readAsArrayBuffer(info.file.originFileObj);
      fileReader.onload = (ev) => {
        try {
          const data = ev.target.result;
          const workbook = XLSX.read(data, { type: "base64" });
          const sheetName = workbook.Sheets[workbook.SheetNames[0]];
          const excelData = XLSX.utils.sheet_to_json(sheetName, {
            header: 1,
            raw: true,
            defval: "",
          });
          // 处理数据
          handleImportData(excelData);
        } catch (error) {
          message({
            type: "warning",
            msg: "文件类型不正确",
          });
        }
        setUploadLoading(false);
        fileReader = null;
      };
    }
    if (status === "error") {
      setUploadLoading(false);
      message({
        type: "error",
        msg: `【${name}】上传失败`,
      });
    }
  },
};
// dom
<Upload
  {...props}
  customRequest={(option) => {
    option.onSuccess();
  }}>
  <Button
    type="primary"
    loading={uploadLoading}>
    导入
  </Button>
</Upload>;
