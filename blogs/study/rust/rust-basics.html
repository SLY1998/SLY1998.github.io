    # 简介

    Rust 是一种系统编程语言，专注于安全性、并发性和性能。它通过独特的所有权系统和借用机制，在编译时就能防止内存安全问题，同时保持高性能。本笔记将带你逐步学习 Rust 的核心概念和基本语法。

    # 安装和配置 Rust 环境

    # 安装 Rust

    在大多数系统上，你可以使用官方提供的 rustup 工具来安装 Rust：

    # 安装 rustup
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh

# 或者使用 PowerShell（Windows）
irm https://sh.rustup.rs -UseBasicParsing | iex

    # 所有权和借用机制

    # 所有权规则

    1. 每个值都有一个所有者
    2. 同一时间只能有一个所有者
    3. 当所有者离开作用域，值将被丢弃
    fn main() {
    let s1 = String::from("hello");
    let s2 = s1;  // s1 的所有权转移给 s2，s1 不再有效
    
    // println!("{}", s1);  // 编译错误：s1 已失去所有权
    println!("{}", s2);     // 正确
}

    # 借用

    使用 & 符号创建引用，允许访问值而不获取所有权：

    fn main() {
    let s1 = String::from("hello");
    let len = calculate_length(&s1);  // 借用 s1
    
    println!("字符串 '{}' 的长度是 {}", s1, len);
}

fn calculate_length(s: &String) -> usize {
    s.len()
}

    # 可变借用

    使用 &mut 创建可变引用，允许修改借用的值：

    change(&mut s);

}

fn change(some_string: &mut String) {
    some_string.push_str(", world");

    # 结构体和枚举

    # 结构体

    // 定义结构体
struct Person {
    name: String,
    age: u32,
    is_student: bool,
}

// 结构体方法
impl Person {
    // 关联函数（类似静态方法）
    fn new(name: String, age: u32, is_student: bool) -> Self {
        Self {
            name,
            age,
            is_student,
        }
    }
    
    // 实例方法
    fn greet(&self) {
        println!("你好，我是 {}", self.name);
    }
    
    // 可变方法
    fn celebrate_birthday(&mut self) {
        self.age += 1;
        println!("生日快乐！我现在 {} 岁了", self.age);
    }
}

// 使用结构体
fn main() {
    let mut person = Person::new(String::from("张三"), 25, false);
    person.greet();
    person.celebrate_birthday();
}

    # 枚举

    // 定义枚举
enum Message {
    Quit,
    Move { x: i32, y: i32 },
    Write(String),
    ChangeColor(i32, i32, i32),
}

// 枚举方法
impl Message {
    fn call(&self) {
        match self {
            Message::Quit => println!("退出消息"),
            Message::Move { x, y } => println!("移动到 ({}, {})", x, y),
            Message::Write(text) => println!("写入消息: {}", text),
            Message::ChangeColor(r, g, b) => println!("改变颜色为 RGB({}, {}, {})", r, g, b),
        }
    }
}

// 使用枚举
fn main() {
    let msg1 = Message::Write(String::from("Hello"));
    let msg2 = Message::Move { x: 10, y: 20 };
    
    msg1.call();
    msg2.call();
}

    # 特质（Trait）

    // 定义特质
pub trait Summary {
    fn summarize(&self) -> String;
    
    // 默认实现
    fn summarize_author(&self) -> String {
        String::from("未知作者")
    }
}

// 实现特质
pub struct Article {
    pub headline: String,
    pub author: String,
    pub content: String,
}

impl Summary for Article {
    fn summarize(&self) -> String {
        format!("{}, by {}", self.headline, self.author)
    }
}

// 使用特质作为参数
pub fn notify<T: Summary>(item: &T) {
    println!("Breaking news! {}", item.summarize());
}

    # 简单项目实战：猜数字游戏

    use std::io;
use std::cmp::Ordering;
use rand::Rng;

fn main() {
    println!("猜数字游戏！");
    
    let secret_number = rand::thread_rng().gen_range(1..=100);
    
    loop {
        println!("请输入你的猜测：");
        
        let mut guess = String::new();
        io::stdin()
            .read_line(&mut guess)
            .expect("无法读取输入");
        
        let guess: u32 = match guess.trim().parse() {
            Ok(num) => num,
            Err(_) => continue,
        };
        
        println!("你猜的是：{}", guess);
        
        match guess.cmp(&secret_number) {
            Ordering::Less => println!("太小了！"),
            Ordering::Greater => println!("太大了！"),
            Ordering::Equal => {
                println!("你赢了！");
                break;
            }
        }
    }
}
